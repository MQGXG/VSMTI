## 第 1 步：快照持久化（0.5 天）

**修改文件**：`packages/core/src/session/snapshot.ts`

**核心变更**：

```typescript
// 1. 修复全局单例 → per-workspace 实例
const snapshotManagers = new Map<string, SnapshotManager>()
export function getSnapshotManager(workspace: string): SnapshotManager {
  let mgr = snapshotManagers.get(workspace)
  if (!mgr) {
    mgr = new SnapshotManager(workspace)
    snapshotManagers.set(workspace, mgr)
  }
  return mgr
}

// 2. 新增磁盘持久化
saveToDisk(): void {
  const dir = path.join(this.workspace, ".mira", "snapshots")
  fs.mkdirSync(dir, { recursive: true })
  const data = this.snapshots.map(s => ({
    id: s.id,
    files: Object.fromEntries(s.files),
    description: s.description,
    timestamp: s.timestamp,
  }))
  fs.writeFileSync(path.join(dir, "snapshots.json"), JSON.stringify(data, null, 2))
}

loadFromDisk(): void {
  try {
    const filePath = path.join(this.workspace, ".mira", "snapshots", "snapshots.json")
    if (!fs.existsSync(filePath)) return
    const data = JSON.parse(fs.readFileSync(filePath, "utf-8"))
    this.snapshots = data.map((s: any) => ({
      ...s,
      files: new Map(Object.entries(s.files)),
    }))
  } catch { /* 静默 */ }
}

// 3. 在 capture/restore/deleteSnapshot 后调用 saveToDisk()
// 4. 构造函数中调用 loadFromDisk()
// 5. 新增 7 天自动清理
cleanup(): void {
  const cutoff = Date.now() - 7 * 24 * 60 * 60 * 1000
  const before = this.snapshots.length
  this.snapshots = this.snapshots.filter(s => new Date(s.timestamp).getTime() > cutoff)
  if (this.snapshots.length < before) this.saveToDisk()
}
```

**向后兼容**：磁盘文件不存在时静默跳过，旧内存快照自动持久化。

export async function searchMessages(query: string): Promise<...> {
  if (!query.trim()) return []
  const db = await getDbAsync()

  // 优先 FTS5
  try {
    const ftsResult = db.exec(
      `SELECT session_id, role, content, snippet(messages_fts, 2, '<mark>', '</mark>', '...', 32) as context
       FROM messages_fts WHERE messages_fts MATCH ? ORDER BY rank LIMIT 50`,
      [query]
    )
    if (ftsResult.length > 0) {
      return ftsResult[0].values.map(row => ({
        session_id: row[0] as string,
        session_title: "",  // 需要 JOIN sessions 表
        message: { role: row[1] as string, content: (row[2] as string).slice(0, 300), timestamp: "" },
        context: row[3] as string,
      }))
    }
  } catch { /* FTS5 不可用，回退 */ }

  // LIKE 回退
  return legacySearch(query)
}
```

**向后兼容**：FTS5 创建失败时自动回退到旧的 LIKE 搜索。sql.js WASM 默认无 FTS5 时走回退路径。

import { IncrementalSummarizer } from "./structured-summary"

export class ContextManager {
  private summarizer = new IncrementalSummarizer()

  // compactPipeline 的 L4 阶段改用结构化摘要
  private async compactHistory(messages: LLMMessage[]): Promise<LLMMessage[]> {
    this.writeTranscript(messages)

    // 更新结构化摘要
    const summary = await this.summarizer.update(messages)

    // 使用结构化格式
    const summaryText = this.summarizer.formatSummary(summary)
    return [{ role: "user" as const, content: `[Compacted]\n\n${summaryText}` }]
  }

  // 溢出触发压缩
  async reactiveCompact(messages: LLMMessage[]): Promise<LLMMessage[]> {
    this.writeTranscript(messages)
    const summary = this.summarizer.getCurrentSummary()
    if (summary) {
      return this.summarizer.overflowCompact(messages, summary, this.config.maxContextTokens)
    }
    return this.legacyReactiveCompact(messages)
  }

  // 可选：设置 LLM 配置以启用增量摘要
  setLLMConfig(config: { apiKey: string; apiUrl: string; model: string; provider: string }): void {
    this.llmConfig = config
    this.checkpointProvider.setLLMConfig(config)
    this.summarizer.setLLMConfig(config)  // 新增
  }

  // 暴露结构化摘要给外部
  getStructuredSummary() { return this.summarizer.getCurrentSummary() }
}
```

**向后兼容**：无 LLM 配置时走旧路径，有 LLM 时自动使用结构化摘要。

## 第 5 步：Event Sourcing 接入（2 天）

**修改文件**：`packages/core/src/session/store.ts`

**核心变更**（双写模式）：

```typescript
import { getEventStore } from "./event-store"
import { getProjector } from "./projector"
import { createMessageEvent } from "./event-types"

const eventStore = getEventStore()
const projector = getProjector()

export async function appendMessage(sessionID: string, message: StoredMessage): Promise<void> {
  // 1. 写入事件层（异步，失败不阻塞）
  try {
    await eventStore.append(createMessageEvent(sessionID, {
      role: message.role,
      content: message.content,
      toolCallId: message.toolCallId,
      retryCount: message.retryCount,
    }))
  } catch { /* 事件层失败不阻塞主流程 */ }

  // 2. 写入旧 messages 表（保持兼容）
  const db = await getDbAsync()
  const existing = db.exec("SELECT title FROM sessions WHERE session_id = ?", [sessionID])
  const isNew = existing.length === 0 || existing[0].values.length === 0
  if (isNew) {
    runWrite("INSERT INTO sessions (session_id, project_id, title, workspace, created_at, updated_at) VALUES (?, '', ?, '', datetime('now'), datetime('now'))",
      [sessionID, `会话 ${new Date().toLocaleDateString("zh-CN")}`])
  }
  runWrite("INSERT INTO messages (session_id, role, content, timestamp, tool_call_id, retry_count) VALUES (?, ?, ?, ?, ?, ?)",
    [sessionID, message.role, message.content, message.timestamp, message.toolCallId || null, message.retryCount || 0])
  runWrite("UPDATE sessions SET updated_at = ? WHERE session_id = ?", [new Date().toISOString(), sessionID])
}

export async function loadSession(sessionID: string): Promise<StoredSession | null> {
  // 优先用 Projector 回放
  try {
    const snapshot = await eventStore.getLatestSnapshot(sessionID)
    const events = await eventStore.getEvents(sessionID, snapshot?.up_to_seq)
    if (events.length > 0 || snapshot) {
      const messages = snapshot
        ? projector.projectFromSnapshot(snapshot, events)
        : projector.replay(events)
      // 从 DB 获取元数据
      const db = await getDbAsync()
      const meta = db.exec("SELECT title, created_at, updated_at, workspace FROM sessions WHERE session_id = ?", [sessionID])
      if (meta.length > 0 && meta[0].values.length > 0) {
        const [title, created, updated, workspace] = meta[0].values[0] as string[]
        return { id: sessionID, title, created, updated, messages, workspace }
      }
    }
  } catch { /* 回放失败 fallback */ }

  // 旧路径
  return legacyLoadSession(sessionID)
}

// 新增：定期保存快照（每 50 个事件）
export async function maybeSaveSnapshot(sessionID: string): Promise<void> {
  const seq = await eventStore.getLatestSeq(sessionID)
  if (seq > 0 && seq % 50 === 0) {
    const messages = await legacyLoadSession(sessionID)
    if (messages) {
      await eventStore.saveSnapshot({
        session_id: sessionID,
        up_to_seq: seq,
        messages_json: JSON.stringify(messages.messages),
        metadata_json: JSON.stringify({ title: messages.title }),
        created_at: new Date().toISOString(),
      })
    }
  }
}
```

**向后兼容**：
- 双写：旧表始终同步写入
- 读取优先 Projector，失败 fallback 到旧路径
- 旧数据库无事件数据时自动走旧路径

**Todo 工具**（todo-tool.ts）：
```typescript
import type { ToolDef } from "../../shared/tool"
import { getDbAsync, runWrite } from "../../system/database"

export const todoTool: ToolDef = {
  name: "todo_write",
  description: "Create, update, list, or complete todo items for task tracking",
  parameters: {
    type: "object",
    properties: {
      action: { type: "string", enum: ["create", "update", "list", "complete", "delete"] },
      id: { type: "string", description: "Todo ID (required for update/complete/delete)" },
      title: { type: "string", description: "Todo title (required for create)" },
      status: { type: "string", enum: ["pending", "in_progress", "done", "cancelled"] },
      priority: { type: "number", description: "Priority (0=low, 1=medium, 2=high)" },
    },
    required: ["action"],
  },
  permission: "todo_write",
  execute: async (args: any, ctx: any) => {
    const db = await getDbAsync()
    const { action, id, title, status, priority } = args

    switch (action) {
      case "create": {
        const todoId = `todo_${Date.now().toString(36)}`
        runWrite("INSERT INTO todos (id, session_id, title, status, priority) VALUES (?, ?, ?, ?, ?)",
          [todoId, ctx.sessionID, title, "pending", priority || 0])
        return { success: true, output: `Created todo: ${todoId}` }
      }
      case "list": {
        const result = db.exec(
          "SELECT id, title, status, priority FROM todos WHERE session_id = ? ORDER BY priority DESC, created_at ASC",
          [ctx.sessionID]
        )
        if (result.length === 0) return { success: true, output: "No todos" }
        const todos = result[0].values.map((r: any) =>
          `[${r[2]}] ${r[1]} (priority: ${r[3]}, id: ${r[0]})`
        ).join("\n")
        return { success: true, output: todos }
      }
      case "update": {
        runWrite("UPDATE todos SET status = ?, priority = ?, updated_at = datetime('now') WHERE id = ? AND session_id = ?",
          [status, priority, id, ctx.sessionID])
        return { success: true, output: `Updated todo: ${id}` }
      }
      case "complete": {
        runWrite("UPDATE todos SET status = 'done', completed_at = datetime('now'), updated_at = datetime('now') WHERE id = ? AND session_id = ?",
          [id, ctx.sessionID])
        return { success: true, output: `Completed todo: ${id}` }
      }
      case "delete": {
        runWrite("DELETE FROM todos WHERE id = ? AND session_id = ?", [id, ctx.sessionID])
        return { success: true, output: `Deleted todo: ${id}` }
      }
      default:
        return { success: false, error: `Unknown action: ${action}` }
    }
  },
}
```
